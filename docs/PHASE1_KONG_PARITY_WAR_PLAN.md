#### Step 2: Create PDF Utility Function (2 hours)
**File:** `client/src/lib/pdfExport.ts`

```typescript
import jsPDF from 'jspdf';
import autoTable from 'jspdf-autotable';

interface PDFConfig {
  title: string;
  subtitle?: string;
  inputs: { label: string; value: string }[];
  sections: { heading: string; content: string | string[] }[];
  footer?: string;
}

export function generatePDF(config: PDFConfig) {
  const doc = new jsPDF();
  let yPosition = 20;

  // Header
  doc.setFontSize(20);
  doc.setFont('helvetica', 'bold');
  doc.text(config.title, 20, yPosition);
  yPosition += 10;

  if (config.subtitle) {
    doc.setFontSize(12);
    doc.setFont('helvetica', 'normal');
    doc.text(config.subtitle, 20, yPosition);
    yPosition += 10;
  }

  // Input Summary Table
  if (config.inputs.length > 0) {
    yPosition += 5;
    autoTable(doc, {
      startY: yPosition,
      head: [['Input Field', 'Value']],
      body: config.inputs.map(input => [input.label, input.value]),
      theme: 'grid',
      headStyles: { fillColor: [139, 92, 246] }, // Purple
    });
    yPosition = (doc as any).lastAutoTable.finalY + 10;
  }

  // Generated Content Sections
  config.sections.forEach((section) => {
    // Section Heading
    doc.setFontSize(14);
    doc.setFont('helvetica', 'bold');
    doc.text(section.heading, 20, yPosition);
    yPosition += 8;

    // Section Content
    doc.setFontSize(10);
    doc.setFont('helvetica', 'normal');
    
    if (Array.isArray(section.content)) {
      section.content.forEach((item, index) => {
        const lines = doc.splitTextToSize(`${index + 1}. ${item}`, 170);
        doc.text(lines, 20, yPosition);
        yPosition += lines.length * 5 + 3;
        
        // Add new page if needed
        if (yPosition > 270) {
          doc.addPage();
          yPosition = 20;
        }
      });
    } else {
      const lines = doc.splitTextToSize(section.content, 170);
      doc.text(lines, 20, yPosition);
      yPosition += lines.length * 5 + 5;
    }

    yPosition += 5;

    // Add new page if needed
    if (yPosition > 270) {
      doc.addPage();
      yPosition = 20;
    }
  });

  // Footer
  if (config.footer) {
    const pageCount = doc.getNumberOfPages();
    for (let i = 1; i <= pageCount; i++) {
      doc.setPage(i);
      doc.setFontSize(8);
      doc.setFont('helvetica', 'italic');
      doc.text(config.footer, 20, 285);
      doc.text(`Page ${i} of ${pageCount}`, 180, 285);
    }
  }

  return doc;
}
```

#### Step 1: Create CharLimitInput Component (1 hour)
**File:** `client/src/components/CharLimitInput.tsx`

```typescript
import { Input } from '@/components/ui/input';
import { Textarea } from '@/components/ui/textarea';
import { Label } from '@/components/ui/label';

interface CharLimitInputProps {
  label: string;
  value: string;
  onChange: (value: string) => void;
  maxLength: number;
  multiline?: boolean;
  placeholder?: string;
  required?: boolean;
}

export function CharLimitInput({
  label,
  value,
  onChange,
  maxLength,
  multiline = false,
  placeholder,
  required = false,
}: CharLimitInputProps) {
  const remaining = maxLength - value.length;
  const isOverLimit = remaining < 0;

  return (
    <div className="space-y-2">
      <Label>
        {label}
        {required && <span className="text-red-500 ml-1">*</span>}
      </Label>
      {multiline ? (
        <Textarea
          value={value}
          onChange={(e) => onChange(e.target.value.slice(0, maxLength))}
          placeholder={placeholder}
          className={isOverLimit ? 'border-red-500' : ''}
        />
      ) : (
        <Input
          value={value}
          onChange={(e) => onChange(e.target.value.slice(0, maxLength))}
          placeholder={placeholder}
          className={isOverLimit ? 'border-red-500' : ''}
        />
      )}
      <p className={`text-sm ${isOverLimit ? 'text-red-500' : 'text-muted-foreground'}`}>
        {value.length}/{maxLength} characters
        {remaining < 10 && remaining >= 0 && (
          <span className="ml-2 text-yellow-500">({remaining} remaining)</span>
        )}
      </p>
    </div>
  );
}
```
